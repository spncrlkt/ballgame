    SELECT level_name, COUNT(*)
    FROM (
        SELECT m.id, m.level_name, SUM(ps.shots_attempted) as shots
        FROM matches m JOIN player_stats ps ON ps.match_id = m.id
        GROUP BY m.id
    )
    WHERE shots <= 5
    GROUP BY level_name
    ORDER BY COUNT(*) DESC
